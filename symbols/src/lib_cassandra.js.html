 21 /**
 22  * @constructor
 23  * A connection to a Cassandra cluster.
 24  *
 25  * @param {String} host A "host:port" pair.
 26  * @api public
 27  *
 28  * @class
 29  * A connection to a Cassandra cluster.
 30  *
 31  * @description
 32  * <p>To connect to a Cassandra server, pass a "host:port" to the constructor then
 33  * use Client#connect.</p>
 34  *
 35  * <p>This is an EventEmitter that you may bind events to.
 36  * See the NodeJS documentation for information on EventEmitter.
 37  * (<a href='http://nodejs.org/docs/v0.4.10/api/all.html#events.EventEmitter'>Link</a>)</p>
 38  *
 39  * @example
 40  *   # Connect to given host and keyspace.
 41  *   var connection = new cassandra.Client("localhost:9160");
 42  *   connection.connect("keyspaceName");
 43  */
 44 var Client = function(host) {
 45   var pair = host.split(/:/);
 46 
 47   /**
 48    * The hostname of the Cassandra server.
 49    * @type String
 50    */
 51   this.host = pair[0];
 52 
 53   /**
 54    * The port number as a string.
 55    * @type String
 56    */
 57   this.port = pair[1];
 58 
 59   /**
 60    * The default consistency level used in transactions.
 61    * @type Object
 62    *
 63    * @description
 64    * This is an object with <code>read</code> and <code>write</code> properties,
 65    * each being a ConsistencyLevel.
 66    */
 67   this.defaultCL = {
 68     read: ttype.ConsistencyLevel.QUORUM,
 69     write: ttype.ConsistencyLevel.QUORUM
 70   };
 71 }
 72 sys.inherits(Client, process.EventEmitter);

 94 /**
 95  * Connects to a Cassandra cluster.
 96  *
 97  * @param {String} keyspace The name of the keyspace to use (optional)
 98  * @param {Object} credentials If given, log into the cluster with the given username and password
 99  * @api public
100  *
101  * @example
102  *   # Connect to `localhost:9160` with the keyspace name `keyspaceName`.
103  *   var connection = new cassandra.Client("localhost:9160");
104  *   connection.connect("keyspaceName");
105  *
106  * @example
107  *   # This example uses the given login credentials.
108  *   var connection = new cassandra.Client("localhost:9160");
109  *   connection.connect("keyspaceName", {user: 'root', password: 'thrift'});
110  *
111  */
112 Client.prototype.connect = function() {
113   var args = Array.prototype.slice.call(arguments);
114 
115   var keyspace_or_credential = args.shift();
116   var credential = args.shift();
117   if (keyspace_or_credential instanceof String ||
118       typeof keyspace_or_credential === 'string') {
119     // if first argument is string, then it is keyspace name
120     this.keyspace = keyspace_or_credential;
121   } else {
122     credential = keyspace_or_credential;
123   }
124 
125   this.ready = false;
126   this.queue = [];
127 
128   this.connection = thrift.createConnection(this.host, this.port);
129   this.connection.on('error', function(err) {
130     this.emit('error', err);
131   });
132   this.thrift_client = thrift.createClient(Cassandra, this.connection);
133 
134   var self = this;
135   this.connection.on('connect', function(err) {
136     if (err) {
137       self.emit('error', err);
138       return;
139     }
140 
141     // if credential is given, then call login
142     if (credential) {
143       self.thrift_client.login(
144         new ttype.AuthenticationRequest(credential), function(err) {
145 
146           if (err) {
147             self.emit('error', err);
148             return;
149           }
150 
151           // only when login is success, emit connected event
152           self.ready = true;
153           self.dispatch();
154         });
155     } else {
156       self.ready = true;
157       self.dispatch();
158     }
159   });
160 
161   // if keyspace is specified, use that ks
162   if (this.keyspace) {
163     this.use(this.keyspace, function(err) {
164       if (err) {
165         self.emit('error', err);
166       }
167     });
168   }
169 };

171 /**
172  * Sets which keyspace to use.
173  *
174  * @param {String} keyspace The name of the keyspace to use
175  * @param {Function} callback (Optional) Callback function to be called after
176  * @api public
177  *
178  * @description
179  * You should probably use {connect} instead.
180  *
181  * @example
182  *   var connection = new cassandra.Client("localhost:9160");
183  *   connection.use("keyspaceName");
184  *   connection.connect();
185  */
186 Client.prototype.use = function(keyspace, callback) {
187   var args = Array.prototype.slice.call(arguments);
188   if (!this.ready) {
189     this.queue.push([arguments.callee, args]);
190     return;
191   }
192 
193   this.keyspace = keyspace;
194 
195   var self = this;
196   this.thrift_client.describe_keyspace(this.keyspace, function(err, ksdef) {
197     if (err) {
198       self.emit('error', err);
199       return;
200     }
201     self.definition_ = ksdef;
202     self.column_families_ = {};
203 
204     var i = ksdef.cf_defs.length;
205     var cf;
206     while (i--) {
207       cf = ksdef.cf_defs[i];
208       self.column_families_[cf.name] = cf;
209     }
210 
211     self.thrift_client.set_keyspace(self.keyspace, function(err) {
212       if (err) {
213         selt.emit('error', err);
214         return;
215       }
216       self.emit('keyspaceSet', self.column_families_);
217     });
218   });
219 };

222  * Set or get default consistency level.
223  *
224  * @param {Object} consistencyLevel (Optional) An object which has write and read consistency level.
225  *           If given, sets default consistency level.
226  * @api public
227  *                         
228  * @description
229  * To set a consistency level, pass a consistencyLevel argument.
230  *
231  * @example
232  *   # Sets the read/write consistency levels
233  *   client.consistencyLevel({
234  *     write: CL.ONE,
235  *     read:  CL.ONE
236  *   });
237  *
238  *   # Gets
239  *   cl = client.consistencyLevel();  //=> { write: CL.ONE, read: CL.ONE }
240  */
241 Client.prototype.consistencyLevel = function() {
242   if (arguments.length == 0) {
243     return this.defaultCL;
244   } else {
245     var newCL = arguments[0];
246     this.defaultCL.read = newCL.read || ttype.ConsistencyLevel.QUORUM;
247     this.defaultCL.write = newCL.write || ttype.ConsistencyLevel.QUORUM;
248   }
249 };

336 /**
337  * @constructor
338  * A column family.
339  *
340  * @param {Client} client the Client
341  * @param {String} name the name of the column family
342  * @api public
343  *
344  * @class
345  * A column family.
346  *
347  * @description
348  * <p>This class can represent either a column family or a super column family. To
349  * check if the object pertains to a super column family, use the isSuper
350  * attribute.</p>
351  *
352  * <p>To retrieve a column family, use Client#getColumnFamily.</p>
353  *
354  * @property {String} name The name of the column family.
355  * @property {String} column_type The type of the column family ('Super' or 'Standard').
356  * @property {Boolean} ready The state of the column family.
357  * @property {Boolean} isSuper True if the column family is a super column family.
358  *
359  * @example
360  *   client = new cassandra.Client("localhost:9160");
361  *   client.connect("keyspaceName");
362  *
363  *   family = client.getColumnFamily('pages');
364  *   family.set(...);
365  *
366  * @see Client#getColumnFamily
367  */
368 var ColumnFamily = function(client, name) {
369   this.name = name;
370   this.queue = [];
371   this.ready = false;
372   this.client_ = client;
373   var self = this;
374   this.client_.on('keyspaceSet', function(cfdef) {
375     // check to see if column name is valid
376     var cf = cfdef[self.name];
377     if (!cf) {
378       // column family does not exist
379       self.client_.emit('error', new Error('Column Family ' + self.name + ' does not exist.'));
380     }
381 
382     // copy all cfdef properties
383     for (var prop in cf) {
384       if (cf.hasOwnProperty(prop)) {
385         self[prop] = cf[prop];
386       }
387     }
388 
389     self.isSuper = self.column_type === 'Super';
390     self.ready = true;
391 
392     self.dispatch();
393   });
394 };

396 /**
397  * Retrieves data for a given key.
398  *
399  * @param keys row keys to fetch
400  * @param columns (optional) which columns to retrieve
401  * @param options (optional) valid params are start, finish, reversed, count
402  * @param {Function} callback Callback function to be called after
403  * @api public
404  *
405  * @description
406  * You may supply options as an object literal. It can have one or more of the
407  * following keys:
408  *
409  *   * start
410  *   * finish
411  *   * reversed (boolean)
412  *   * count (number)
413  *   * consistencyLevel (ConsistencyLevel)
414  *
415  */
416 ColumnFamily.prototype.get = function() {
417   var args = Array.prototype.slice.call(arguments);
418   if (!this.ready) {
419     this.queue.push([arguments.callee, args]);
420     return;
421   }
422 
423   // last argument may be callback
424   var callback;
425   if (typeof args[args.length - 1] === 'function') {
426     callback = args.pop();
427   }
428   // keys to get
429   var keys = args.shift();
430   // if only one key specified, turn it to array
431   if (!(keys instanceof Array)) {
432     keys = [keys];
433   }
434 
435   var method_args = this.isSuper ? this.parseArgumentsForSuperCF_(args) : this.parseArgumentsForStandardCF_(args);
436   var column_parent = method_args[0];
437   var predicate = method_args[1];
438   var cl = method_args[2] || this.client_.defaultCL.read;
439 
440   var self = this;
441   this.client_.thrift_client.multiget_slice(
442     keys, column_parent, predicate, cl,
443     function(err, res) {
444       if (err) {
445         callback(err, obj);
446       }
447 
448       // array -> obj
449       var obj = {};
450       var key, col, sub_col;
451       for (key in res) {
452         if (res.hasOwnProperty(key)) {
453           obj[key] = {};
454           var i = res[key].length;
455           while (i--) {
456             col = res[key][i].super_column;
457             if (col) {
458               // super
459               obj[key][col.name] = {};
460               var j = col.columns.length;
461               while (j--) {
462                 sub_col = col.columns[j];
463                 obj[key][col.name][sub_col.name] = sub_col.value;
464               }
465             } else {
466               // standard
467               col = res[key][i].column;
468               obj[key][col.name] = col.value;
469             }
470           }
471         }
472       }
473       if (keys.length == 1) {
474         obj = obj[keys[0]];
475       }
476       callback(err, obj);
477     });
478 };

544 /**
545  * Sets data through insert or delete.
546  *
547  * @param {String} key
548  * @param {Object} values
549  * @param {Object} options
550  * @param {Function} callback (Optional) Callback function to be called after
551  * @api public
552  *
553  * @description
554  * For super column families, values should be 2 levels down.
555  *
556  * @example
557  *   # Sets some data for a normal column family.
558  *   names = client.getColumnFamily('names');
559  *   names.set("fruits", { "a": "apple" });
560  *
561  * @example
562  *   # Sets some data for a super column family.
563  *   metrics = client.getColumnFamily('metrics');
564  *   if (metrics.isSuper) {
565  *     metrics.set(
566  *       "august_2011",
567  *       {
568  *         "10": { visits: 29, pageViews: 84 },
569  *         "11": { visits: 14, pageViews: 29 }
570  *       });
571  *   }
572  */
573 ColumnFamily.prototype.set = function() {
574   var args = Array.prototype.slice.call(arguments);
575   if (!this.ready) {
576     this.queue.push([arguments.callee, args]);
577     return;
578   }
579 
580   var callback;
581   if (typeof args[args.length - 1] === 'function') {
582     callback = args.pop();
583   }
584 
585   var key = args.shift();
586   var values = args.shift() || {};
587   var options = args.shift() || {};
588   var cl = options.consistencyLevel || this.client_.defaultCL.write;
589   var ts = new Date().getTime();
590 
591   var prop, value;
592   var mutations = [], columns;
593   if (this.isSuper) {
594     // super
595     for (prop in values) {
596       if (values.hasOwnProperty(prop)) {
597         columns = [];
598         value = values[prop];
599         for (var col in value) {
600           columns.push(new ttype.Column({
601             name: col,
602             value: '' + value[col],
603             timestamp: ts,
604             ttl: null
605           }));
606         }
607         // prop is super column name
608         mutations.push(new ttype.Mutation({
609           column_or_supercolumn: new ttype.ColumnOrSuperColumn({
610             super_column: new ttype.SuperColumn({
611               name: prop,
612               columns: columns
613             })
614           })
615         }));
616       }
617     }
618   } else {
619     // standard
620     for (prop in values) {
621       mutations.push(new ttype.Mutation({
622         column_or_supercolumn: new ttype.ColumnOrSuperColumn({
623           column: new ttype.Column({
624             name: prop,
625             value: '' + values[prop],
626             timestamp: ts,
627             ttl: null
628           })
629         })
630       }));
631     }
632   }
633 
634   var mutation_map = {};
635   mutation_map[key] = {};
636   mutation_map[key][this.name] = mutations;
637 
638   this.client_.thrift_client.batch_mutate(mutation_map, cl, callback);
639 };

722 /**
723  * 
724  * @api private
725  * @param args
726  * @return [ColumnParent, SlicePredicate, ConsistencyLevel]
727  */
728 ColumnFamily.prototype.parseArgumentsForSuperCF_ = function(args) {
729   var default_options = {
730     start: '',
731     finish: '',
732     reversed: false,
733     count: 100,
734     consistencyLevel: null
735   };
736   var column_parent = {
737     column_family: this.name
738   };
739   var predicate = {};
740 
741   var super_column_or_options = args.shift();
742   var options = default_options;
743 
744   if (super_column_or_options) {
745     // first argumet may be super column name
746     if (super_column_or_options instanceof String ||
747         typeof super_column_or_options === 'string') {
748       column_parent.super_column = super_column_or_options;
749       var columns_or_options = args.shift();
750       if (columns_or_options) {
751         var columns, options, option_name;
752         if (typeof columns_or_options.slice === 'function') {
753           // first argument is column name(s)
754           columns = columns_or_options.slice();
755           if (!(columns instanceof Array)) {
756             columns = [columns];
757           }
758           predicate.column_names = columns;
759           options = args.shift() || default_options;
760         } else {
761           // update default option with given value
762           for (option_name in columns_or_options) {
763             if (columns_or_options.hasOwnProperty(option_name)) {
764               options[option_name] = columns_or_options[option_name];
765             }
766           }
767           predicate.slice_range = new ttype.SliceRange(options);
768         }
769       } else {
770         predicate.slice_range = new ttype.SliceRange(options);
771       }
772     } else {
773       // update default option with given value
774       for (option_name in super_column_or_options) {
775         if (super_column_or_options.hasOwnProperty(option_name)) {
